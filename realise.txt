SELECT 
	*,modelos.id as id_modelo
FROM 
	marcas_carros as marcas 
LEFT JOIN 
	modelos_carros as modelos ON modelos.id_marca = marcas.id 			
WHERE
	modelos.id is NULL 	
ORDER BY marcas.nome, modelos.nome asc LIMIT 10000
